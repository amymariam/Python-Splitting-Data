np.random.seed(42)

ages = []
for ii in range(250):
    ages.append(random.randint(18,75))
    
net_worths = [ii * 6.25+np.random.normal(scale=40) for ii in ages]

ages = np.reshape(np.array(ages),(len(ages),1))

net_worths = np.reshape(np.array(net_worths),(len(net_worths),1))

from sklearn.model_selection import train_test_split

ages_train, ages_test, net_worths_train, net_worths_test = train_test_split(ages,net_worths)

reg1 = studentReg(ages_train,net_worths_train)

print("Coefficient=", reg1.coef_)
print("Intercept=", reg1.intercept_)
print("Training data score=", reg1.score(ages_train,net_worths_train))
print("Testing data score=", reg1.score(ages_test,net_worths_test))
